func findMaximumXOR(nums []int) int {
    root := &Trie{}
    res := 0
    for i := 1; i < len(nums); i++ {
        root.insert(nums[i-1])
        res = max(res, root.check(nums[i]))
    }
    return res
}

// 构造前缀树--左子树为0, 右子树为1
type Trie struct {
    left, right *Trie
}

func (t *Trie) insert(num int) {
    cur := t
    for i := HighBit; i >= 0; i-- {
        // 查看当前的比特
        bit := num>>i&1
        // 如果当前位是1,走右子树
        if bit == 1 {
            if cur.right == nil {
                cur.right = &Trie{}
            }
            cur = cur.right
        } else {
            // 走左子树
            if cur.left == nil {
                cur.left = &Trie{}
            }
            cur = cur.left
        }
    }
}

// 求当前数能在前缀树中找到的最异或大值
func (t *Trie) check(num int) int {
    cur := t
    x := 0
    for i := HighBit; i >= 0; i-- {
        bit := num>>i&1
        // 当前位置是1, 走左子树, 去找0
        if bit == 1 {
            if cur.left != nil {
                x = x*2+1
                cur = cur.left
            } else {
                x = x*2
                cur = cur.right
            }
        } else {
            // 当前位置为0, 走右子树, 去找1
            if cur.right != nil {
                x = x*2+1
                cur = cur.right
            } else {
                x = x*2
                cur = cur.left
            }
        }
    }
    return x
}

const HighBit = 30

func max(a, b int) int {
    if a > b {
        return a
    }
    return b
}
