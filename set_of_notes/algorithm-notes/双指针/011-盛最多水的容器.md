func maxArea(height []int) int {
    // 双指针
    N := len(height)
    left, right := 0, N-1 
    maxArea := 0
    for left < right {
        length := right - left // 长
        high := Min(height[left], height[right])
        maxArea = Max(maxArea, high*length)
        if height[left] <= height[right] {
            left++
        } else {
            right--   
        }
    }
    return maxArea
}

func Max(a ...int) int {
    max := a[0]
    for _, v := range a[1:] {
        if v > max { max = v }
    }
    return max
}

func Min(a ...int) int {
    min := a[0]
    for _, v := range a[1:] {
        if v < min { min = v }
    }
    return min
}
