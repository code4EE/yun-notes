#### [46. 全排列 (中等)](https://leetcode-cn.com/problems/permutations/)

## 解答

```go
func permute(nums []int) [][]int {
    res := [][]int{}

    var dfs func(int)
    dfs = func(index int) {
        if index == len(nums) {
            res = append(res, append([]int(nil), nums...))
            return
        }
        for i := index; i < len(nums); i++ {
            // 做出选择
            swap(nums, i, index)
            
            dfs(index+1)
            // 撤销选择
            swap(nums, i, index)
        }
    }
    dfs(0)
    return res
}

func swap(a []int, i, j int) {
    a[i], a[j] = a[j], a[i]
}
```
