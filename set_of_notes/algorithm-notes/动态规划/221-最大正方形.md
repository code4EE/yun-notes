#### [221. 最大正方形](https://leetcode-cn.com/problems/maximal-square/)

## 解答

```go
// 正序, dp[i][j] 为以matrix[i][j]为右下角的正方形的最大边长
func maximalSquare(matrix [][]byte) int {
    M, N := len(matrix), len(matrix[0])
    dp := make([][]int, M)
    for i := range dp {
        dp[i] = make([]int, N)
    }
    res := 0
    for i := 0; i < M; i++ {
        for j := 0; j < N; j++ {
            dp[i][j] = int(matrix[i][j]-'0')
            if dp[i][j] == 1 {
                res = max(res, 1)
            }
            if i > 0 && j > 0 {
                if matrix[i][j] == '0' {
                    dp[i][j] = 0
                } else {
                    dp[i][j] = min(dp[i][j-1], dp[i-1][j], dp[i-1][j-1]) + 1
                    if dp[i][j] > res {
                        res = dp[i][j]
                    }
                }
            }
        }
    }
    return res*res
}

func min(a ...int) int {
    ans := a[0]
    for _, v := range a[1:] {
        if v < ans {
            ans = v
        }
    }
    return ans
}

func max(a, b int) int {
    if a > b {
        return a
    }
    return b
}
```
