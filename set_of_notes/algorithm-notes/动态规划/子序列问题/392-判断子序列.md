## 解答

##### 双指针

```go
// 最简单的--双指针
func isSubsequence(s string, t string) bool {
    lenS, lenT := len(s), len(t)
    i, j := 0, 0 
    for i < lenS && j < lenT {
        if s[i] == t[j] {
            i++
        }
        j++
    }
    return i == lenS
}
```



##### 动态规划

```go
// 代码和【不同的子序列】一样
func isSubsequence(s string, t string) bool {
    // 
    M, N := len(t), len(s)
    dp := make([][]int, M+1)
    for i := range dp {
        dp[i] = make([]int, N+1)
    }
    for i := 0; i <= M; i++ {
        dp[i][0] = 1
    }
    for i := 1; i <= M; i++ {
        for j := 1; j <= N; j++ {
            if t[i-1] == s[j-1] {
                dp[i][j] = dp[i-1][j-1] + dp[i-1][j]
            } else {
                dp[i][j] = dp[i-1][j]
            }
        }
    }
    // 查看最后一列是否为0, 这里直接查最后一个了
    if dp[M][N] != 0 {
        return true
    }
    return false
}
```
