## 解答

```go
func buildTree(preorder []int, inorder []int) *TreeNode {
    // base case
    if len(preorder) == 0 {
        return nil
    }
    // 找到根节点
    rootVal := preorder[0]
    preorder = preorder[1:]
    root := &TreeNode{Val: rootVal}
    // 在中序遍历中找到根节点
    i := 0
    for ; i < len(inorder); i++ {
        if inorder[i] == rootVal {
            break
        }
    }
    // 从中序遍历中找到左右子树的遍历序列
    left  := inorder[:i]
    right := inorder[i+1:]
    root.Left = buildTree(preorder[:len(left)], left)
    root.Right = buildTree(preorder[len(left):], right)
    return root
}
```
